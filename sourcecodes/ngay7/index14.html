Vòng lặp 
        for (statement 1; statement 2; statement 3) {
            // code thực thi trong vòng lặp 
        }
        statement 1 => var i = 1; // khởi tạo giá trị cho biến đếm 
        statement 2 => i <= 10; // điều kiện dừng vòng lặp 
        statement 3 => i++ // sau mỗi lần lặp thì i tăng thêm 1 đơn vị 
        i++ tương đường i = i + 1

        // chú ý : vòng lặp vô hạn treo trình duyệt
